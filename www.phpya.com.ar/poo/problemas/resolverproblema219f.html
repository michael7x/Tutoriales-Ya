class Cuadrado {
  private $lado;
  public function cargarLado($la)
  {
    $this->lado=$la;
  }
  public function retornarPerimetro()
  {
    $p=$this->lado*4;
    return $p;
  }
  public function retornarSuperficie()
  {
    $s=$this->lado*$this->lado;
    return $s;
  }
}
