var x;
x=$(document);
x.ready(inicializarEventos);

function inicializarEventos()
{
  var x=$("#boton1");
  x.click(hipervinculoGoogle);
  x=$("#boton2");
  x.click(hipervinculoYahoo);
  x=$("#boton3");
  x.click(hipervinculoMsn);
}

function hipervinculoGoogle()
{
  var x=$("a");
  x.attr("href","http://www.google.com");
  x.text("Google");
}

function hipervinculoYahoo()
{
  var x=$("a");
  x.attr("href","http://www.yahoo.com");
  x.text("Yahoo");
}

function hipervinculoMsn()
{
  var x=$("a");
  x.attr("href","http://www.msn.com");
  x.text("Msn");
}

Definir un hipervínculo con la propiedad href con cadena vacía. Luego disponer tres botones que permitar fijar distintos hipervínculos para la propiedad href. Además actualizar el texto del hipervínculo.
