var x;
x=$(document);
x.ready(inicializarEventos);

function inicializarEventos()
{
  var x=$("#boton1");
  x.click(asociarClase);
  x=$("#boton2");
  x.click(desasociarClase);
}

function asociarClase()
{
  var x=$("th");
  x.addClass("titulo");
  var x=$("td");
  x.addClass("cuerpo");
}

function desasociarClase()
{
  var x=$("th");
  x.removeClass("titulo");
  var x=$("td");
  x.removeClass("cuerpo");
}
